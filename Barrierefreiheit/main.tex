    \item[\tipp] Texte als \textbf{Textseite} anlegen (HTML oft besser als PDF)
    \item[\ja]   Dokumente als \textbf{.docx} oder \textbf{.pdf} bereitstellen
    \item[\ja]   Bilder in Kursen und H5P mit kurzem \textbf{Alternativtext}
    \item[\ja]   H5P: \textbf{Semantische Überschriften} mit struktureller
                 Bedeutung -- nicht nur Aussehen
    \item[\warn] Autoplay/autom. Übergänge \textbf{deaktivieren}
    \item[\tipp] Nutzungshinweise (Bedienung/Tastatur) voranstellen
  }%
}{%
  \themenbox{\AIcon{Video-Symbol}{\faVideo}}{Videos \& Live/Hybrid}{%
    \item[\ja]   \textbf{Untertitel} aktivieren und nachbearbeiten
    \item[\ja]   Video \textbf{pausierbar}, Lautstärke einstellbar
    \item[\ja]   Wichtige Bildinfos \textbf{gesprochen erklären}
    \item[\tipp] \textbf{Transkript/Notizen} (2--5 Stichpunkte)
    \item[\tipp] Live: \textbf{Mikro nutzen}, Fragen wiederholen,
                 kurz zusammenfassen
  }%
}{%
  \themenbox{\AIcon{Clipboard mit Liste}{\faClipboardList}}{Aufgaben/Tests \& Tools}{%
    \item[\ja]   Aufgaben: \textbf{Kriterien + Dateiformat} klar nennen

    \item[\warn] Drag\,\&\,Drop vermeiden oder\allowbreak{} Alternative bereitstellen
    \item[\ja]   Feedback: Text statt nur Farbe

  }%

  \themenbox{\AIcon{Tafel-Symbol}{\faChalkboard}}{Präsentationen (PowerPoint)}{%

    \item[\ja]   \textbf{Farbe + Symbol} (nicht Farbe allein)
    \item[\ja]   \textbf{Kontrast} Schrift\,/\,Hintergrund:
                 mind.\,4{,}5:1 (groß: 3:1)
    \item[\ja]   \textbf{Lesereihenfolge} prüfen

    \item[\warn] Keine automatischen Übergänge/Animationen
  }%
}{%
  \themenbox{\AIcon{Bild-Symbol}{\faImage}}{Bilder \& Alternativtexte}{%
    \item[\ja]   Inhaltliche Bilder: \textbf{Alternativtext} eintragen
    \item[\ja]   Dekorative Bilder: als ,,dekorativ`` markieren
    \item[\ja]   Alternativtext: Aussage + Kontext (,,Diagramm zeigt \ldots``)

  }%
